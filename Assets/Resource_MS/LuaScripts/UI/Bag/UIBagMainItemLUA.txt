function UIBagMainItemLUA:OnCloseMe()
	if(self.view~=nil)then
		if(self.view.isUIMonoItem)then
			self:RemoveItem(self);
		else
			self:CloseUI("UIBagMainItem")
		end
	end
end
--当UI主动或被动被关闭时会执行，此时界面不会销毁
function UIBagMainItemLUA:OnPause()
	self:RemoveUIEvent()
	self:RemoveEventListener()
end
--当UI被主动关闭并且缓存时间到被销毁时执行
function UIBagMainItemLUA:OnDestroy()
	UIBagMainItemLUA = nil;
end
