function MainUILUA:OnCloseMe()
	if(self.view~=nil)then
		if(self.view.isUIMonoItem)then
			self:RemoveItem(self);
		else
			self:CloseUI("MainUI")
		end
	end
end
function MainUILUA:OnDestroy()
	self:RemoveEventListener()
	MainUILUA = nil;
end
