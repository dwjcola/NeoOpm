LoginProxy=LUAUIBase:new({
	});
function LoginProxy:Init()
	
end
function LoginProxy:EnterServer(ServerInfo)
	local EnterServerReq = {}
	EnterServerReq.AccId = ServerInfo.AccId;
	self:Warning("RPCLUA.wfGate_Handler.EnterServer")
	RPCLUA.wfGate_Handler.EnterServer(EnterServerReq,self.EnterServerBack,self)
end
function LoginProxy:EnterServerBack(rep)
	self:Warning("RPCLUA.wfGate_Handler.EnterServer   back")
	self:SendEvent("EnterServerBack",rep)
end
--登录返回信息
function LoginProxy:UpdataPlayerData(info)
	PlayerData:UpdateInfo(info.PlayerBaseInfo)
end
return LoginProxy
